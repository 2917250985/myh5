### 2.3 Switch语句

#### 2.3.1 语法

~~~JavaScript
switch(st){
    case 'st':   
        执行代码块
         break
    case 'st+n':   
        执行代码块
        break
   	default:
        执行代码块
        break
        
}
~~~

#### 2.3.2 例子

~~~javascript
let num1 = +prompt('请输入第一个数：')
let st = prompt('请输入运算符+-*/其中一个：')
let num2 = +prompt('请输入第二个数：')
switch (st) {
    case '+':
        document.write(num1 + num2)
        break
    case '-':
        document.write(num1 - num2)
        break
    case '*':
        document.write(num1 * num2)
        break
    case '/':
        document.write(num1 / num2)
        break
    // 不符合条件执行
    default:
        document.write('输入错误')
        break
}
~~~
